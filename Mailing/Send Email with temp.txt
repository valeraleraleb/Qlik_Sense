  call StringDecoder($(item));
  set $(item)_temp = $(EncodeString);
  //let temp_var = '$(item)' & '_temp';
  //let $(temp_var) = $(EncodeString);
  //Trace $(EncodeString);
  //Trace $(temp_var) = $([$(temp_var)]);
  
next

    status as SendEmail_status,
